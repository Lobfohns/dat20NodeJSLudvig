Server Side Rendering to allow the applicaton to use the public folder.


    app.use(express.static("public"));

How to combine multiple html pages to achieve Server Side Rendering


    // fs stands for FileSystem
    const fs = require("fs");

    const nav = fs.readFileSync("./public/components/nav/nav.html").toString();

    const footer = fs.readFileSync("./public/components/footer/footer.html".toString());

    const frontpage = fs.readFileSync("./public/pages/frontpage/frontpage.html").toString();

    const frontpagePage = nav + frontpage + footer;

Basic get request


    app.get("/", (req, res) => {
        res.send(frontpagePage);
    });
